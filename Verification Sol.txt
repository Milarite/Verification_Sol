        function checkHashExsist(string _hast) public view returns (bool)
        {
            for(uint i=0;i<hash.length;i++)
            {
                if(keccak256(abi.encodePacked(hash[i]))==keccak256(abi.encodePacked(_hast)))
                {
                return true;
                break;
                }
            }
            return false;
        }
