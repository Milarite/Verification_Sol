pragma solidity ^0.4.24;
import './Ownable.sol';
contract Authorizable is Ownable {
    
    mapping (address => bool) isAuth ;
    address [] AuthorizableAddress;
    
    function addAuthUser (address _address) public onlyOwner {
        isAuth[_address]  =true;
        AuthorizableAddress.push(_address);
    }
    
    function isAuthUser ()public view returns (bool){
        return isAuth[msg.sender];
    }
    
     modifier onlyAuth() {
    require(isAuthUser());
    _;
  }
    
    
}
